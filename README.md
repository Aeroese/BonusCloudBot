# BonusCloudBot
博纳云收益报时机器人

#安装
pip install pyTelegramBotAPI
git clone https://github.com/Aeroese/BonusCloudBot.git

#配置Config.ini
Token = telegram bot token
Email = BonusCloud User
Password = BonusCloud Pwd
Chatid = Your telegram chatid
Time = 6:30

#运行
screen -S bxc
cd BonusCloudBot
python3 bot.py
